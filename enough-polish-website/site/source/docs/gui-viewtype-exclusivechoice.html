The exclusive-choice view-type shows one option of an exclusive ChoiceGroup at a time.

Design

.exclusiveChoice {
	label-style: choiceGroupLabel;
	background-type: round-rect;
	background-color: focusedBgColor;
	include-label: true;
	layout: expand;
	view-type: exclusive-choice;
	exclusiveview-left-arrow: url( arrow-left.png );
	exclusiveview-right-arrow: url( arrow-right.png );
}

.exclusiveChoice:hover {
	background-color: white;
}

}
